\newglossaryentry{record}{
    name=record,
    description={In Fluent bit, a record is specifically referring to a key-value pair. An event can be made up of 1 or more records}
}
\newglossaryentry{event}{
    name=event,
    description={In Fluent bit, an event is a set of key-value pairs with an associated timestamp and tag. Events are routed according to Fluent Bit's configured internal dataflow graph}
}

\newglossaryentry{tag}{
    name=tag,
    description={In Fluent bit, a tag is associated with every event and can be set or matched through Wildcard or Regex patterns. Tags serve as the primary means to route events within Fluent Bit}
}
